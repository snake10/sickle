163  static QVector3D calcTriNormal(QVector3D t) {
164  auto V = vertex(t.y()) - vertex(t.x());
165  auto W = vertex(t.z()) - vertex(t.x());
166  QVector3D normal(

168  (V.z() * W.x()) - (V.x() * W.z()),

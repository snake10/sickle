257  static void calcNormals() {
258  Child::s_normals = Child::s_vertices;
259  Child::s_tangents = Child::s_vertices;
260  Child::s_bitangents = Child::s_vertices;
261  auto size = Child::s_indices.size();
262  for(int i = 0; i < size; i += 3) {
263  auto tri = triangle(i);
264  auto v = calcTriNormal(tri);
265  auto tan = calcTriTangent(tri);
266  auto btan = calcTriBitangent(tri);
267  auto t = tri * 3;
268  for(int j = 0; j < 3; j++) {
269  Child::s_normals[t.x() + j] = Child::s_normals[t.y() + j] = Child::s_normals[t.z() + j] = v[j];
270  Child::s_tangents[t.x() + j] = Child::s_tangents[t.y() + j] = Child::s_tangents[t.z() + j] = tan[j];
271  Child::s_bitangents[t.x() + j] = Child::s_bitangents[t.y() + j] = Child::s_bitangents[t.z() + j] = btan[j];

276  static QVector3D calcTriNormal(QVector3D tri) {
277  auto t = tri * 3;
278  auto V = vertex(t.y()) - vertex(t.x());
279  auto W = vertex(t.z()) - vertex(t.x());
280  QVector3D normal(
281  (V.y() * W.z()) - (V.z() * W.y()),
282  (V.z() * W.x()) - (V.x() * W.z()),
283  (V.x() * W.y()) - (V.y() * W.x()));
284  normal.normalize();
285  return normal;
286  }

288  static QVector3D calcTriTangent(QVector3D tri) {
289  auto verT = tri * 3;
290  auto V = vertex(verT.y()) - vertex(verT.x());
291  auto W = vertex(verT.z()) - vertex(verT.x());
292  auto uvT = tri * 2;
293  auto UV1 = UV(uvT.y()) - UV(uvT.x());
294  auto UV2 = UV(uvT.z()) - UV(uvT.x());
295  auto r = 1.0f / (UV1.x() * UV2.y() - UV1.y() * UV2.x());
296  return (V * UV2.y() - W * UV1.y()) * r;
297  }

299  static QVector3D calcTriBitangent(QVector3D tri) {
300  auto verT = tri * 3;
301  auto V = vertex(verT.y()) - vertex(verT.x());
302  auto W = vertex(verT.z()) - vertex(verT.x());
303  auto uvT = tri * 2;
304  auto UV1 = UV(uvT.y()) - UV(uvT.x());
305  auto UV2 = UV(uvT.z()) - UV(uvT.x());
306  auto r = 1.0f / (UV1.x() * UV2.y() - UV1.y() * UV2.x());
307  auto btan = (W * UV1.x() - V * UV2.x()) * r;
308  return btan;

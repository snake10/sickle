247  static void calcNormals() {
248  auto normals = Child::s_buffersData.value("Position");
249  auto tangents = normals;
250  auto bitangents = normals;
251  auto size = Child::s_indexBuffer.size();
252  for(int i = 0; i < size; i += 3) {
253  auto tri = triangle(i);
254  auto v = calcTriNormal(tri);
255  auto tan = calcTriTangent(tri);
256  auto btan = calcTriBitangent(tri);
257  auto t = tri * 3;
258  for(int j = 0; j < 3; j++) {
259  normals[t.x() + j] = normals[t.y() + j] = normals[t.z() + j] = v[j];
260  tangents[t.x() + j] = tangents[t.y() + j] = tangents[t.z() + j] = tan[j];
261  bitangents[t.x() + j] = bitangents[t.y() + j] = bitangents[t.z() + j] = btan[j];
262  }
263  }

266  Child::s_buffersData.insert("Tangent", tangents);

270  static QVector3D calcTriNormal(QVector3D tri) {
271  auto t = tri * 3;
272  auto V = vertex(t.y()) - vertex(t.x());
273  auto W = vertex(t.z()) - vertex(t.x());
274  QVector3D normal(
275  (V.y() * W.z()) - (V.z() * W.y()),
276  (V.z() * W.x()) - (V.x() * W.z()),
277  (V.x() * W.y()) - (V.y() * W.x()));
278  normal.normalize();
279  return normal;
280  }

282  static QVector3D calcTriTangent(QVector3D tri) {
283  auto verT = tri * 3;
284  auto V = vertex(verT.y()) - vertex(verT.x());
285  auto W = vertex(verT.z()) - vertex(verT.x());
286  auto uvT = tri * 2;
287  auto UV1 = UV(uvT.y()) - UV(uvT.x());
288  auto UV2 = UV(uvT.z()) - UV(uvT.x());
289  auto r = 1.0f / (UV1.x() * UV2.y() - UV1.y() * UV2.x());
290  return (V * UV2.y() - W * UV1.y()) * r;
291  }

293  static QVector3D calcTriBitangent(QVector3D tri) {
294  auto verT = tri * 3;
295  auto V = vertex(verT.y()) - vertex(verT.x());
296  auto W = vertex(verT.z()) - vertex(verT.x());
297  auto uvT = tri * 2;

299  auto UV2 = UV(uvT.z()) - UV(uvT.x());
300  auto r = 1.0f / (UV1.x() * UV2.y() - UV1.y() * UV2.x());
301  auto btan = (W * UV1.x() - V * UV2.x()) * r;

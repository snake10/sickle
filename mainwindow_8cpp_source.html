00060 QVector3D* toEuler(const QQuaternion& quat) {
00061     qreal pitch, yaw, roll;
00062     const auto q = quat.toVector4D();
00063     const auto w2 = q.w() * q.w();
00064     const auto x2 = q.x() * q.x();
00065     const auto y2 = q.y() * q.y();
00066     const auto z2 = q.z() * q.z();
00067     const auto unitLength = w2 + x2 + y2 + z2;
00068     const auto abcd = q.w() * q.x() + q.y() * q.z();
00069     const auto eps = 1e-7;
00070     if (abcd > (0.5 - eps) * unitLength) {
00071         yaw = 2 * qAtan2(q.y(), q.w());
00072         pitch = M_PI;
00073         roll = 0;
00074     } else if (abcd < (-0.5 + eps) * unitLength) {
00075         yaw = -2 * qAtan2(q.y(), q.w());
00076         pitch = -M_PI;
00077         roll = 0;
00078     } else {
00079         const auto adbc = q.w() * q.z() - q.x() * q.y();
00080         const auto acbd = q.w() * q.y() - q.x() * q.z();
00081         yaw = qAtan2(2 * adbc, 1 - 2 * (z2 + x2));
00082         pitch = qAsin(2 * abcd / unitLength);
00083         roll = qAtan2(2 * acbd, 1 - 2 * (y2 + x2));
00084     }
00085     return new QVector3D(qRadiansToDegrees(pitch), qRadiansToDegrees(yaw), qRadiansToDegrees(roll));

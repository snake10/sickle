00054 QVector3D* toEuler(const QQuaternion& quat) {
00055     float heading, attitude, bank;
00056     auto q1 = quat.toVector4D();
00057     Q_ASSERT(q1.x() == quat.vector().x());
00058     Q_ASSERT(q1.y() == quat.vector().y());
00059     Q_ASSERT(q1.z() == quat.vector().z());
00060     Q_ASSERT(q1.w() == quat.scalar());
00061     double test = q1.x() * q1.y() + q1.z() * q1.w();
00062     if (test > 0.499) {
00063         heading = 2 * atan2(q1.x(), q1.w());
00064         attitude = M_PI / 2;
00065         bank = 0;
00066     } else if (test < -0.499) {
00067         heading = -2 * atan2(q1.x(), q1.w());
00068         attitude = - M_PI / 2;
00069         bank = 0;
00070     } else {
00071         double sqx = q1.x() * q1.x();
00072         double sqy = q1.y() * q1.y();
00073         double sqz = q1.z() * q1.z();
00074         heading = atan2(2 * q1.y() * q1.w() - 2 * q1.x() * q1.z(), 1 - 2 * sqy - 2 * sqz);
00075         attitude = asin(2 * test);
00076         bank = atan2(2 * q1.x() * q1.w() - 2 * q1.y() * q1.z(), 1 - 2 * sqx - 2 * sqz);
00077     }
00078     return new QVector3D(heading, attitude, bank);
00079 }

116 QQuaternion fromEuler(const QVector3D& euler) {
117  auto value = QQuaternion::fromAxisAndAngle(QVector3D(1, 0, 0), euler.x());
118  value *= QQuaternion::fromAxisAndAngle(QVector3D(0, 1, 0), euler.y());
119  value *= QQuaternion::fromAxisAndAngle(QVector3D(0, 0, 1), euler.z());
120  value.normalize();

124 QVector3D* toEuler(const QQuaternion& quat) {
125  qreal pitch, yaw, roll;
126  const auto q = quat.toVector4D();
127  const auto w2 = q.w() * q.w();
128  const auto x2 = q.x() * q.x();
129  const auto y2 = q.y() * q.y();
130  const auto z2 = q.z() * q.z();
131  const auto unitLength = w2 + x2 + y2 + z2;
132  const auto abcd = q.w() * q.x() + q.y() * q.z();
133  const auto eps = std::numeric_limits<float>::epsilon();
134  if (abcd > (0.5 - eps) * unitLength) {
135  yaw = 2 * qAtan2(q.y(), q.w());
136  pitch = M_PI;
137  roll = 0;
138  } else if (abcd < (-0.5 + eps) * unitLength) {
139  yaw = -2 * qAtan2(q.y(), q.w());
140  pitch = -M_PI;
141  roll = 0;
142  } else {
143  const auto adbc = q.w() * q.z() - q.x() * q.y();
144  const auto acbd = q.w() * q.y() - q.x() * q.z();
145  yaw = qAtan2(2 * adbc, 1 - 2 * (z2 + x2));
146  pitch = qAsin(2 * abcd / unitLength);
147  roll = qAtan2(2 * acbd, 1 - 2 * (y2 + x2));
148  }
149  return new QVector3D(qRadiansToDegrees(pitch), qRadiansToDegrees(yaw), qRadiansToDegrees(roll));
150 }

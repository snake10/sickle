00039 QQuaternion fromEuler(const QVector3D& euler) {
00040     double c1 = cos(euler.x() / 2);
00041     double s1 = sin(euler.x() / 2);
00042     double c2 = cos(euler.y() / 2);
00043     double s2 = sin(euler.y() / 2);
00044     double c3 = cos(euler.z() / 2);
00045     double s3 = sin(euler.z() / 2);
00046     double c1c2 = c1 * c2;
00047     double s1s2 = s1 * s2;
00048     auto w = c1c2 * c3 - s1s2 * s3;
00049     auto x = c1c2 * s3 + s1s2 * c3;
00050     auto y = s1 * c2 * c3 + c1 * s2 * s3;
00051     auto z = c1 * s2 * c3 - s1 * c2 * s3;
00052     auto angle = 2 * acos(w);
00053     double norm = x * x + y * y + z * z;
00054     if (norm < 0.001) {
00055         x = 1;
00056         y = z = 0;
00057     } else {
00058         norm = sqrt(norm);
00059         x /= norm;
00060         y /= norm;
00061         z /= norm;
00062     }

221 void Viewport::renderQuad() {
222  glBindVertexArray(m_quadVAO);
223  glBindFramebuffer(GL_FRAMEBUFFER, defaultFramebufferObject());
224  glDisable(GL_DEPTH_TEST);
225 
226  glClearColor(m_bgColor.redF(), m_bgColor.greenF(), m_bgColor.blueF(), m_bgColor.alphaF());
227  glClear(GL_COLOR_BUFFER_BIT);
228 
229  glEnable(GL_BLEND);
230  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
231 
232  auto status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
233  if(status != GL_FRAMEBUFFER_COMPLETE) {
234  qWarning() << "Main buffer status:" << status;
235  return;
236  }
237 
238  if(m_showBuffers || m_showMaps) {
239  auto prog = m_quadPrograms.value(0);
240  prog->bind();
241 
242  int len;
243  QList<Light*> lights;
244  if(m_showBuffers) {
245  len = m_sceneTextures.length();
246  } else {
247  lights = m_world->currentZone()->findChildren<Light*>();
248  len = lights.length();
249  }
250 
251  if(len < 1) {
252  glViewport(0, 0, width(), height());
253  } else {
254  int div = 1, w, h;
255  while(pow(div, 2) < len + 1) div++;
256  w = width() / div;
257  h = height() / div;
258 
259  for(int i = 0; i <= len; i++) {
260  int x = i % div,
261  y = (div - 1) - (i / div);
262  glViewport(x * w, y * h, w, h);
263  if(i < len) {
264  glActiveTexture(GL_TEXTURE0);
265  if(m_showBuffers) {
266  glBindTexture(GL_TEXTURE_2D, m_sceneTextures.at(i));
267  } else {
268  glBindTexture(GL_TEXTURE_2D, lights.at(i)->getTexture());
269  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_COMPARE_MODE, GL_NONE);
270  }
271  glEnableVertexAttribArray(0);
272  glBindBuffer(GL_ARRAY_BUFFER, m_quadBuffer);
273  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr);
274  glDrawArrays(GL_TRIANGLES, 0, 6);
275  glDisableVertexAttribArray(0);
276  if(m_showMaps)
277  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_COMPARE_MODE, GL_COMPARE_REF_TO_TEXTURE);
278  }
279  }
280  prog->release();
281  }
282  } else {
283  glViewport(0, 0, width(), height());
284  }
285 
286  int lightsLen = 1;
287  QList<Light*> lightList;
288  bool isLit = m_program == tr("Light");
289  if(isLit) {
290  lightList = m_world->currentZone()->findChildren<Light*>();
291  lightsLen = lightList.length() + 1;
292  }
293 
294  for(int i = 0; i < lightsLen; i++) {
295  bool isAmbient = isLit && i == 0;
296  bool isLight = isLit && !isAmbient;
297 
298  int id;
299  if (isAmbient)
300  id = 2;
301  else if (isLight)
302  id = 1;
303  else
304  id = 0;
305 
306  auto prog = m_quadPrograms.value(id);
307  prog->bind();
308 
309  Light* light = nullptr;
310  if(isLight)
311  light = lightList.value(i - 1);
312 
313  static const QList<QString> names = {
314  "color",
315  "normal",
316  "tangent",
317  "bitangent",
318  "vertPos",
319  "matProp1",
320  "matProp2",
321  "matProp3",
322  "depth",
323  "shadowMap"
324  };
325  int maxLen = m_sceneTextures.length();
326  int texLen = 1;
327  if(isLit) texLen = maxLen;
328  if(isLight) texLen++;
329  for(int j = 0; j < texLen; j++) {
330  glActiveTexture(GL_TEXTURE0 + j);
331  auto tex = (isLight && j >= maxLen) ? light->getTexture() : m_sceneTextures.at(j);
332  glBindTexture(GL_TEXTURE_2D, tex);
333  auto n = names.at(j);
334  auto loc = prog->uniformLocation(n);
335  if(loc > -1)
336  prog->setUniformValue(loc, j);
337 #ifdef DEBUG_SHADERS // Les uniforms sont souvent supprimés lors de l'optimisation des shaders durant leur compilation
338  else
339  qWarning().noquote() << "Uniform" << n << "not found.";
340 #endif
341  }
342 
343 
344  if(isLit) {
345  if(isAmbient) {
346  prog->setUniformValue("ambientColor", toVector(m_ambient));
347  auto occlusion = qobject_cast<AmbientOcclusion*>(m_world->AO());
348  prog->setUniformValue("AO.threshold", occlusion->threshold());
349  prog->setUniformValue("AO.kernelSize", QVector2D(
350  occlusion->kernelSize() / width(),
351  occlusion->kernelSize() / height()));
352  prog->setUniformValue("AO.maxDist", occlusion->maxDist());
353  } else {
354  auto type = light->type();
355  prog->setUniformValue("light.type", type);
356  prog->setUniformValue("light.orientation", light->direction());
357  prog->setUniformValue("light.location", light->position());
358  prog->setUniformValue("light.power", light->power());
359  prog->setUniformValue("light.color", toVector(light->color()));
360 
361  if(type == 1) {
362  auto spot = static_cast<Spotlight*>(light);
363  prog->setUniformValue("light.falloff", spot->farZ());
364  prog->setUniformValue("light.inner", static_cast<GLfloat>(qCos(qDegreesToRadians(spot->innerAngle()))));
365  prog->setUniformValue("light.outer", static_cast<GLfloat>(qCos(qDegreesToRadians(spot->outerAngle()))));
366  }
367 
368  prog->setUniformValue("vDepth", light->view());
369  prog->setUniformValue("pDepth", light->projection());
370  prog->setUniformValue("eyeD", -m_camera->direction().normalized());
371  }
372  }
373 
374  glEnableVertexAttribArray(0);
375  glBindBuffer(GL_ARRAY_BUFFER, m_quadBuffer);
376  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr);
377  glDrawArrays(GL_TRIANGLES, 0, 6);
378  glDisableVertexAttribArray(0);
379 
380  prog->release();
381 
382  if(isAmbient) {
383  glBlendEquation(GL_FUNC_ADD);
384  glBlendFunc(GL_ONE, GL_ONE);
385  }
386  }
387 
388  glDisable(GL_BLEND);
389  glEnable(GL_DEPTH_TEST);
390 }
